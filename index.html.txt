<script>
  // --- Fun√ß√£o de digita√ß√£o ---
  function typeText(el, text, speed = 30) {
    return new Promise((resolve) => {
      let i = 0;
      const interval = setInterval(() => {
        el.textContent = text.substring(0, i);
        i++;
        if (i > text.length) {
          clearInterval(interval);
          el.style.borderRight = "none";
          resolve();
        }
      }, speed);
    });
  }

  // --- Pop-up de carregamento ---
  async function showLoading() {
    const { value: fake } = await Swal.fire({
      title: "CARREGANDO...",
      html: `
        <div class="loading-bar" id="loading-bar">
          ${Array(10).fill('<div class="bar-block"></div>').join('')}
        </div>
        <button class="btn" disabled>Cancelar</button>
      `,
      showConfirmButton: false,
      allowOutsideClick: false,
      didOpen: () => animateBar()
    });
  }

  // --- Anima√ß√£o da barrinha ---
  async function animateBar() {
    const blocks = document.querySelectorAll(".bar-block");
    for (let j = 0; j < 3; j++) {
      for (let i = 0; i < blocks.length; i++) {
        blocks[i].style.opacity = "1";
        await new Promise(r => setTimeout(r, 100));
      }
      for (let b of blocks) b.style.opacity = "0.2";
    }
    showQuestion();
  }

  // --- Pop-up da pergunta ---

  }
